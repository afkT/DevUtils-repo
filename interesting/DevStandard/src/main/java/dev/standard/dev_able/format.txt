package dev.utils.common.able;

/**
 * detail: 通用 AAAAA 接口
 * @author Ttt
 * <pre>
 *     所有通用快捷 AAAAA 接口定义存储类
 *     全部接口只定义一个方法 BBBBB() 且返回值一致
 *     唯一差异就是参数不同CCCCC
 * </pre>
 */
public final class AAAAAable {

    private AAAAAable() {
    }

    // =======
    // = 无参 =
    // =======

    public interface AAAAA<T> {

        T BBBBB();
    }

    // =======
    // = 有参 =
    // =======

    public interface AAAAAByParam<T, Param> {

        T BBBBB(Param param);
    }

    public interface AAAAAByParam2<T, Param, Param2> {

        T BBBBB(
                Param param,
                Param2 param2
        );
    }

    public interface AAAAAByParam3<T, Param, Param2, Param3> {

        T BBBBB(
                Param param,
                Param2 param2,
                Param3 param3
        );
    }

    public interface AAAAAByParamArgs<T, Param> {

        T BBBBB(Param... args);
    }
}
